### **2. Responsive Grids:**

```tsx
// Bad âŒ

  {items.map(...)}
</div>

// Good âœ…
<div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4 md:gap-6">
  {items.map(...)}
</div>

  {items.map(...)}
</div>
```

```tsx

        <MobileLayout />

        <DesktopLayout />
      )}
    </>
  );
}
```
